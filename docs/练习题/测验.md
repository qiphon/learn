2. 写出输出值，为什么？

```js

this.a = 20;

function go(){
    console.log(this.a);
    this.a = 30
}
go.prototype.a = 40

var test = {
    a: 50,
    init: function(fn){
        fn()
        console.log(this.a)
        return fn;
    }
}

console.log((new go()).a)  // 30   

// 上面答案  40  30   构造函数实例化时会执行一次，30 是 new go().a 的结果

test.init(go)  // 50  30  

// 上面答案  20  50
// 因为调用 go的 this 指向 window，相当于 test.init(window.go)

var p = test.init(go)  // 

p();  // 20

// 答案  30  上面 go方法修改的

```
