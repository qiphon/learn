```js
function debounce(cb, time= 1000){
    var timeout = null;
    return function (){
        if(timeout){
            clearTimeout(timeout)
            timeout = null
        }
        timeout = setTimeout(function(){
            cb()
        }, time)
    }
}

function throttle(cb, time=1000){
    var timeout = null;
    return function(){
        if(timeout) return;
        timeout = setTimeout(function(){
            timeout = null
            cb()
        },time)
    }
}

```
